def snt(i) 
    for p in range(2,i) :
        if (i%p == 0) : return False
    return True

t = int(input())
for i in range(1,t + 1) :
    n = int(input())
    listsnt = []
    for i in range(2, n) :
        if (snt(i) == True) : listsnt.append(i)
    for u in range(1, len(listsnt) - 2) :
        tichsnt = listsnt[u] * listsnt[u + 1]
        if (tichsnt <= n) :
            kq = n - tichsnt
        else : break
    print(int(kq))
